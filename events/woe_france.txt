country_event = {
	id = woe_fra_elections.2
	title = woe_fra_elections.2.title
	desc = woe_fra_elections.2.desc
	picture = GFX_report_event_generic_parliament
	is_triggered_only = yes
	
	option = {
		name = woe_fra_elections.2.a
		trigger = {
			original_tag = FRA
			democratic > 0.55
			communism < 0.3
		}
		

		FRA = {
			set_country_flag = FRA_leader_lebrun_active
		}
		
		if = {
			limit = {
				OR = {
					has_stability  < 0.3
					monarchist > 0.3
				}
			}
			start_civil_war = {
				ideology = monarchist
				size = 0.3
			}
			else = {
				
				news_event = { id = woe_fra_news.2 hours = 12 }
			}	
		}

	}

	option = {
		name = woe_fra_elections.2.b
		trigger = {
			original_tag = FRA
			democratic < 0.3
			communism > 0.55
		}
		set_politics = {
			ruling_party = communism
			elections_allowed = no
		} 
		if = {
			limit = {
				OR = {
					has_stability < 0.3
					monarchist > 0.3
				}
			}
			start_civil_war = {
				ideology = monarchist
				size = 0.3
			}	
			else = {
				news_event = { id = woe_fra_news.3 hours = 12 }
			}
		}
	}
	option = {
		name = woe_fra_elections.2.c  # A Divided Nation
		trigger = {
			original_tag = FRA
			communism > 0.3
			democratic > 0.3
		}
		add_stability = -0.15
		add_political_power = -100
	
		# Optional: Add an idea to represent instability
		add_ideas = national_deadlock
	
		# Monarchists exploit the chaos if strong enough
		if = {
			limit = {
				OR = {
					has_stability < 0.3
					monarchist > 0.3
				}
			}
			start_civil_war = {
				ideology = monarchist
				size = 0.3
			}
		}
		else = {
			news_event = { id = woe_fra_news.3 hours = 12 }
		}
	}
	
}
